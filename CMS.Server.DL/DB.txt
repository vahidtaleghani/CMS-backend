    contracttypeid integer NOT NULL,

        ON DELETE NO ACTION
        NOT VALID,
    CONSTRAINT contractstatusid FOREIGN KEY (contractstatusid)

        NOT VALID,
    CONSTRAINT contracttypeid FOREIGN KEY (contracttypeid)
